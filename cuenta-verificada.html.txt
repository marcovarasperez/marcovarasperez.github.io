<script src="https://cdn.jsdelivr.net/npm/bootstrap@5.3.0/dist/js/bootstrap.bundle.min.js"></script>
<script>
    const API_BASE = "https://marcovarasperez.duckdns.org/api/jugadores";
    const params   = new URLSearchParams(window.location.search);
    const ok       = params.get('ok') === 'true';

    // Usuario pendiente (puede venir de sessionStorage si llegan desde verificar-email)
    const usuarioPendiente = sessionStorage.getItem('rpg_usuario_pendiente');

    function showToast(msg, type) {
        const wrap = document.getElementById('toast-wrap');
        const t = document.createElement('div');
        t.className = 'toast-msg ' + (type || 'ok');
        t.innerHTML = `<span>${type === 'err' ? '✖' : '✦'}</span><span>${msg}</span>`;
        wrap.appendChild(t);
        setTimeout(() => { t.style.animation = 'slideOut .3s forwards'; setTimeout(() => t.remove(), 300); }, 3200);
    }

    if (ok) {
        document.getElementById('view-ok').style.display = 'block';
        sessionStorage.removeItem('rpg_email_pendiente');
        sessionStorage.removeItem('rpg_usuario_pendiente');

        // Cuenta atrás y redirección automática
        let seg = 5;
        const iv = setInterval(() => {
            seg--;
            document.getElementById('countdown').textContent = seg;
            if (seg <= 0) { clearInterval(iv); window.location.href = 'index.html'; }
        }, 1000);

    } else {
        document.getElementById('view-error').style.display = 'block';

        // Si no hay usuario en sessionStorage, ocultar botón de reenvío
        if (!usuarioPendiente) {
            document.getElementById('btn-resend-error').style.display = 'none';
        }
    }

    // ── Reenviar desde página de error ───────────────────────────────────────
    const COOLDOWN = 30;
    let resendCooldown = false;

    async function reenviarDesdeError() {
        if (resendCooldown || !usuarioPendiente) return;
        resendCooldown = true;

        const btn  = document.getElementById('btn-resend-error');
        const bar  = document.getElementById('cooldown-bar-error');
        const fill = document.getElementById('cooldown-fill-error');

        btn.disabled = true;
        btn.innerHTML = '<span class="spinner"></span> &nbsp; Enviando…';

        try {
            const res = await fetch(`${API_BASE}/reenviar-verificacion/${encodeURIComponent(usuarioPendiente)}`, {
                method: 'POST'
            });
            if (res.ok) {
                showToast('Correo reenviado. Revisa tu bandeja de entrada.', 'ok');
                // Llevar de vuelta a verificar-email para que haga polling
                setTimeout(() => { window.location.href = 'verificar-email.html'; }, 1800);
            } else {
                const err = await res.text();
                showToast(err || 'Error al reenviar el correo.', 'err');
            }
        } catch (e) {
            showToast('No se pudo conectar con el servidor.', 'err');
        }

        // Cooldown visual
        bar.style.display = 'block';
        fill.style.width = '100%';
        let restante = COOLDOWN;
        const iv = setInterval(() => {
            restante--;
            fill.style.width = (restante / COOLDOWN * 100) + '%';
            btn.innerHTML = `↺ &nbsp; Reenviar (${restante}s)`;
            if (restante <= 0) {
                clearInterval(iv);
                btn.disabled = false;
                btn.innerHTML = '↺ &nbsp; Enviar nuevo correo de verificación';
                bar.style.display = 'none';
                fill.style.width = '100%';
                resendCooldown = false;
            }
        }, 1000);
    }

    // ── Partículas ────────────────────────────────────────────────────────────
    (function() {
        const c = document.getElementById('particles'), ctx = c.getContext('2d');
        let W, H, pts = [];
        const r = () => { W = c.width = innerWidth; H = c.height = innerHeight; };
        window.addEventListener('resize', r); r();
        for (let i = 0; i < 40; i++) pts.push({ x:Math.random()*W, y:Math.random()*H, r:Math.random()*1.1+.3, spd:Math.random()*.2+.05, d:(Math.random()-.5)*.09 });
        (function f() {
            ctx.clearRect(0, 0, W, H);
            pts.forEach(p => {
                ctx.beginPath(); ctx.arc(p.x, p.y, p.r, 0, Math.PI*2);
                ctx.fillStyle = `rgba(201,168,76,${.15+.45*Math.abs(Math.sin(Date.now()*.001+p.x))})`; ctx.fill();
                p.y -= p.spd; p.x += p.d;
                if (p.y < -5) { p.y = H+5; p.x = Math.random()*W; }
                if (p.x < 0 || p.x > W) p.d *= -1;
            });
            requestAnimationFrame(f);
        })();
    })();
</script>
